redis cache tester

# 1. check redis is running
docker exec -it redis redis-cli ping
expected: PONG
# 2. check plugin connection status via wp-cli
docker exec -it wordpress wp redis status --allow-root --path=/var/www/html
expected: Status: Connected
# 3. check key count before browsing the site
docker exec -it redis redis-cli dbsize
browse the site, then run again — the number should increase
# 4. list all keys stored by wordpress
docker exec -it redis redis-cli keys "*"
shows all cached wordpress objects
# 5. flush the cache
docker exec -it wordpress wp redis flush --allow-root --path=/var/www/html
clears all redis keys — run dbsize after to confirm it drops to 0
# 6. verify WP_REDIS_HOST is set in wp-config
docker exec -it wordpress grep REDIS /var/www/html/wp-config.php
should show WP_REDIS_HOST and WP_REDIS_PORT definitions
